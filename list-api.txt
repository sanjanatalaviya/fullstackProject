6. Find the user name with the maximum likes of posts.
[
  {
    $unwind: {
      path: "$posts"
    }
  },
  {
    $group: {
      "_id": "$_id",
      name: { $first: "$name" },
      totallike: { $sum: "$posts.likes" }
    }
  },
